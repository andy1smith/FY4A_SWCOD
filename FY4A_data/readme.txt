'time': UTC timestamp
'C01~C06': SW reflectance
'C08~C14': brightness temperature [K]
'C08_rad~C14_rad': radiance [W/(m^2 sr cm^-1)]
'Sat_Azi', 'Sat_Zen': SatelliteAzimuth, SatelliteZenith
'Sun_Azi', 'Sun_Gli', 'Sun_Zen': SunAzimuth, SunGlintAngle, SunZenith
'ele':  elevation
'T_a': air temperature (2-m above ground) [K]
'RH': relative humidity (2-m above ground) [%]
'ghi': GHI [w/m2]
